	var init = function () {
		var submitAsyncRequest = function(type, url, callBack, postData){
			var d = new Date();
			var bigNumber = " " + d.getYear() + d.getDay() + d.getHours() + d.getMinutes() + d.getSeconds() + d.getMilliseconds();
			if(postData == ""){
				postData = 'random=' + (Math.floor(Math.random() * bigNumber));
			}else{
				postData += '&random=' + (Math.floor(Math.random() * bigNumber));
			}
			if(type == 'POST'){
				YAHOO.util.Connect.asyncRequest(type, url, callBack, postData);
			}else{
				YAHOO.util.Connect.asyncRequest(type, url, callBack);
			}
		}

		var receivedRefreshedCaptchaCallback = {
			cache:false,
			success: function(o) {
				if(o.responseXML){
					if(o.responseXML.getElementsByTagName("error")[0].firstChild.nodeValue == 1 || o.responseXML.getElementsByTagName("error")[0].firstChild == null){
						alert("There has been an error retrieving your new question.  We apologize for the inconvenience.");
					} else {
						var newQuestion;
						var newCaptchaQuestionXML = o.responseXML;
						if (newCaptchaQuestionXML.getElementsByTagName("question").length > 0) {
							document.getElementById("reportCaptcha").innerHTML = newCaptchaQuestionXML.getElementsByTagName("question")[0].firstChild.nodeValue;
						}
					}
				}
			},
			failure: function(o) {}
		};

		function refreshCaptcha(p_oEvent) {
			submitAsyncRequest('POST', '/common/servlet/CommonController', receivedRefreshedCaptchaCallback, 'op=9857');
		}


		// Limit the number of characters at submit time that the user can right click->paste in the report job comments textarea.
		function captchaSubmitClick(p_oEvent) {
			limitChars();
			document.reportJob.submit();
		}

		YAHOO.util.Event.onContentReady("bd", function() {
			var oLinkButton1 = new YAHOO.widget.Button("captchaRefresh", { onclick: { fn: refreshCaptcha } });
			var oLinkButton2 = new YAHOO.widget.Button("captchaSubmit", { onclick: { fn: captchaSubmitClick } });
			var oLinkButton3 = new YAHOO.widget.Button("rateJob");
			var oLinkButton4 = new YAHOO.widget.Button("joinNetwork");
			YAHOO.util.Event.addListener("rateJob", "click", rateJobClicked);
		});

		// Handle a rate button click on the originating page. Set up some fields for later use and show the panel.
		// NOTE: This is when the rate button is clicked from within the YUI dropdown, not the rate button that displays the dropdown.
		function rateJobClicked(p_oEvent) {
			YAHOO.util.Event.stopEvent(p_oEvent);
		
			var r = Math.floor(Math.random()*10000);
			for (i = 1; i <= 3; i++) {
				var ratingID = "rating" + i;
				if (document.getElementById(ratingID).checked) {
					var rating = document.getElementById(ratingID).value;
				}
			}

			document.location.href = "/jobsearch/servlet/JobSearch?op=306&dockey=xml/9/f/9f2f048647455f7b962b1899160a8aba@endecaindex&source=19&currdoc=1&queryString=&r=" + r + "&rating=" + rating;
		
		}
	}();

	// Limit the number of characters the user can enter in the report job comments textarea.
	function limitChars() {
		var text1 = document.getElementById('reportExplain');
		var max = 400;
		if(text1.value.length >= max) {
			text1.value = text1.value.substring(0, max - 1);
		}
	}

    function clearText(textField) {
        textField.value = "";
    }

    // User clicked the login button on the panel. Send request to server for authentication.
    function logMeIn(p_oEvent) {
        var sjt_user_name = document.getElementById("SJT_USER_NAME").value;
        var sjt_password = document.getElementById("SJT_PASSWD").value;
        var xmlhttp = null;
        if (window.XMLHttpRequest) {
            // code for Firefox, Opera, IE7, etc.
            xmlhttp = new XMLHttpRequest();
        }
        else if (window.ActiveXObject) {
            // code for IE6, IE5
            xmlhttp = new ActiveXObject("Microsoft.XMLHTTP");
        }
        if (xmlhttp != null) {
            // Generate random to prevent caching.
            var d = new Date();
            var bigNumber = " " + d.getYear() + d.getDay() + d.getHours() + d.getMinutes() + d.getSeconds() + d.getMilliseconds();
            var randomNumber = (Math.floor(Math.random() * bigNumber));

            xmlhttp.onreadystatechange = logMeInResponded(xmlhttp, processLogMeInResponse);
            // Note: The dockey may not be populated. If it is populated the report and rate job information is included in the response.
            xmlhttp.open("GET", "http://seeker.dice.com/profman/servlet/ProfMan?op=2000&ajax=1&SUBMIT=1&SJT_USER_NAME=" + sjt_user_name + "&SJT_PASSWD=" + sjt_password + "&dockey=xml/9/f/9f2f048647455f7b962b1899160a8aba@endecaindex" + "&random=" + randomNumber, true);
            xmlhttp.send(null);
        }
        else {
            //alert("Your browser does not support XMLHTTP.");
        }
    }

    // Server responded to out request for authentication. Check status codes.
    function logMeInResponded(xmlhttp, responseXmlHandler) {
        return function () {

            if (xmlhttp.readyState == 4) {
                // 4 = "loaded"
                if (xmlhttp.status == 200) {
                    // 200 = "OK"
                    //alert("xml: " + xmlhttp.responseText);
                    responseXmlHandler(xmlhttp.responseXML);
                }
                else {
                    //alert("Problem retrieving data:" + xmlhttp.statusText);
                }
            }
        }
    }

    // Process the server's response to our request for authentication.
    function processLogMeInResponse(locations) {
        var locationsVals = locations.getElementsByTagName("locations")[0];
        // If we didn't get anything back from the server,
        // exit here or risk the wrath of IE's stellar javascript error detection.
        if (locationsVals == null) {
            return;
        }
        var items = locationsVals.getElementsByTagName("item");

        for (var I = 0; I < items.length; I++) {
            var item = items[I];
            validLogin = item.getElementsByTagName("validLogin")[0].firstChild.nodeValue;

            if (validLogin == 1) {
                //var loginPanel = new YAHOO.widget.Panel("loginPanelDiv").hide();

                document.body.className = "yui-skin-sam";

                var jobRating;
                var applicationDate;
                var jobInfoPresent = item.getElementsByTagName("jobInfoPresent")[0].firstChild.nodeValue;
                if (jobInfoPresent == "1") {
                    if (item.getElementsByTagName("jobRating")[0].firstChild != null) {
                        jobRating = item.getElementsByTagName("jobRating")[0].firstChild.nodeValue;
                    }
                    if (item.getElementsByTagName("applicationDate")[0].firstChild != null) {
                        applicationDate = item.getElementsByTagName("applicationDate")[0].firstChild.nodeValue;
                    }
                    if (item.getElementsByTagName("jobReportedDate")[0].firstChild != null) {
                        jobReportedDate = item.getElementsByTagName("jobReportedDate")[0].firstChild.nodeValue;
                    }
                }

                var matchJobAlerts = item.getElementsByTagName("matchJobAlert")[0].firstChild.nodeValue;

            
            
            
                if (submitAfterLogin == 1) {
                    // Call the appropriate processor based on which button was clicked to bring up the login dialog in the first place.
                    document.location.href = document.getElementById("destinationLink").value;
                }
                else {
                    // We're staying on this page. Update any information that needs to be updated.
                    loginPanel.hide();
                    isAuthenticated = "true";

                    
                        document.getElementById("rating1").checked = false;
                        document.getElementById("rating2").checked = false;
                        document.getElementById("rating3").checked = false;
                        if (jobRating == 1) {
                            document.getElementById("rating1").checked = true;
                        }
                        if (jobRating == 0 || jobRating == 99) {
                            document.getElementById("rating2").checked = true;
                        }
                        if (jobRating == -1) {
                            document.getElementById("rating3").checked = true;
                        }
                    

                    // Rate This Job was clicked. Populate the rate job dropdown with seeker's rating information for
                    // this job before showing the rating dropdown.
                    if (submitAfterLogin == 2) {
                        rateJobPanelShow();
                    }
                    else if (submitAfterLogin == 3) {
                        // Report This Job was clicked. Determine which dropdown to show based on whether this seeker has
                        // previously reported this job.
                        if (jobReportedDate == null || jobReportedDate == "") {
                            reportJobPanelShow();
                        }
                        else {
                            reportJobPanelReportedShow();
                        }
                    } else if (submitAfterLogin == 4) {
                        // Join Network was clicked. Or maybe connect to company or recruiter
                        document.location.href = document.getElementById("destinationLink").value;
                    } else if (submitAfterLogin == 6) {

                        if(matchJobAlerts == "true") {
                            matchJobAlertPanelShow();
                            matchJobAlertInfoPanelHide();
                        } else {
                            matchJobAlertPanelHide();
                            matchJobAlertInfoPanelShow();
                            //Hide the form information
                            document.getElementById("InfoMessageText").innerHTML = "You have reached your maximum number of search agents.";
                            document.getElementById("").style.display = "none";
                        }
                }
                }
            
            
            }
            else {
                // AUthorization failed. Inform the user.
                var error = item.getElementsByTagName("error")[0].firstChild.nodeValue;
                var loginError = document.getElementById("loginError");
                loginError.innerHTML = error.replace("\n", "<br>");
            }
        }
    }

    // Catch the click event and determine if we've clicked outside of the report or rate job panels.
    // If so, hide the panel.
    document.onclick=checkClick;
    function checkClick(e) {
        var target = (e && e.target) || (event && event.srcElement);

    
        var obj = document.getElementById('rateJobPanel');
        checkParent(target, "rateJobPanel") ? rateJobPanelHide() : null;
    

    
        obj = document.getElementById('reportJobPanel');
        checkParent(target, "reportJobPanel") ? reportJobPanelHide() : null;
    

    

    

    
        obj = document.getElementById('reportJobPanelReported');
        checkParent(target, "reportJobPanelReported") ? reportJobPanelReportedHide() : null;
    

    
        var obj = document.getElementById('companyNetworkPanel');
        checkParent(target, "companyNetworkPanel") ? connectPanelHide() : null;
    
    }

    // Used by checkClick method to crawl up parental elements from the one clicked looking for the panel div name.
    function checkParent(target, panelDiv) {
        if (document.getElementById(panelDiv) != null) {
        }
        while(target.parentNode) {
            if (target == document.getElementById(panelDiv)) {
                return false;
            }
            target = target.parentNode;
        }
        return true;
    }
